#---------------------- CA --------------------------

# Подготовка каталогов

cd ~/openssl_certs/rootCA
mkdir certs crl newcerts private
chmod 700 private
touch index.txt
echo 1000 > serial

# Создание корневого ключа 4096

cd ~/openssl_certs/rootCA
openssl genrsa -aes256 -out private/ca.key.crt 4096

# Создание корневого сертификата

cd ~/openssl_certs/rootCA
openssl req -config openssl_rootCA.cnf \
 -key private/ca.key.crt \
 -new -x509 -days 7305 -sha256 -extensions v3_ca \
 -out certs/ca.cert.crt


# Проверка корневого сертификата
openssl x509 -noout -text -in certs/ca.cert.crt

#------------- Intermediate CA (ImCA) ---------------

# Подготовка каталогов
cd ~/cert
mkdir ./imCA
cd ~/opensslCA/imCA
mkdir certs crl csr newcerts private
chmod 700 private
touch index.txt
echo 1000 > serial
echo 1000 > crlnumber

# Создание промежуточного ключа 4096

cd ~/opensslCA/imCA
openssl genrsa -aes256 -out private/intermediate.key.crt 4096

# Создание промежуточного сертификата

cd ~/opensslCA/imCA
openssl req -config openssl_imCA.cnf -new -sha256 \
 -key private/intermediate.key.crt \
 -out csr/intermediate.csr.crt

openssl ca -config openssl_imCA.cnf -extensions v3_intermediate_ca \
-days 3652 -notext -md sha256 \
-in  csr/intermediate.csr.crt \
-out certs/intermediate.cert.crt

# Создание файла цепочки сертификатов

cd ~/opensslCA/imCA
cat certs/intermediate.cert.crt ~/opensslCA/rootCA/certs/ca.cert.crt > certs/ca-chain.cert.crt
chmod 444 certs/ca-chain.cert.crt

# Проверка промежуточного сертификата

cd ~/opensslCA/imCA

openssl x509 -noout -text -in certs/intermediate.cert.crt

openssl verify -CAfile ~/opensslCA/rootCA/certs/ca.cert.crt ~/opensslCA/imCA/certs/intermediate.cert.crt

Утилита update-ca-certificates обновляет файл /etc/ssl/certs/ca-certificates.crt 
на основе сертификатов, расположенных в /usr/local/share/ca-certificates/ 
При добавлении сертификат должен быть в формате PEM и иметь расширение '.crt'
Доверенные сертификаты хранятся в каталоге /etc/ssl/certs/ 

sudo update-ca-certificates

Для проверки просмотреть содержимое /etc/ssl/certs/ca-certificates.crt который содержит все доверенные сертификаты. 



openssl s_client -showcerts -connect nas.lan

В Chrome есть специальная страница для просмотра и управления доверенными центрами сертификации. Вот как её открыть:

Откройте браузер Chrome.
Нажмите на значок меню в правом верхнем углу окна (обычно это три точки, иногда три горизонтальные полоски).
В выпадающем меню выберите пункт "Настройки".
Прокрутите страницу вниз и нажмите на ссылку "Дополнительные настройки" (обычно она находится внизу страницы).
Прокрутите страницу вниз до раздела "Безопасность" и выберите "Управление сертификатами".
Откроется окно с настройками сертификатов. В нем выберите вкладку "Центры сертификации".
В этой вкладке нажмите на кнопку "Доверенные центры".
